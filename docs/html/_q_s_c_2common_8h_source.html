609#if defined(QSC_SYSTEM_NATIVE_UINT128)
610# define QSC_SYSTEM_FAST_64X64_MUL(X,Y,Low,High) \
611 do { \
612 const uint128_t r = (uint128_t)(X) * (Y); \
613 *(High) = (r >> 64) & 0xFFFFFFFFFFFFFFFFULL; \
614 *(Low) = (r) & 0xFFFFFFFFFFFFFFFFULL; \
615 } while(0)
616#elif defined(QSC_SYSTEM_COMPILER_MSC) && defined(QSC_SYSTEM_IS_X64)
617# include <intrin.h>
618# pragma intrinsic(_umul128)
619# define QSC_SYSTEM_FAST_64X64_MUL(X,Y,Low,High) \
620 do { \
621 *(Low) = _umul128((X), (Y), (High)); \
622 } while(0)
623#elif defined(QSC_SYSTEM_COMPILER_GCC)
624# if defined(QSC_SYSTEM_ARCH_IX86)
625# define QSC_SYSTEM_FAST_64X64_MUL(X,Y,Low,High) \
626 do { \
627 asm("mulq %3" : "=d" (*(High)), "=X" (*(Low)) : "X" (X), "rm" (Y) : "cc"); \
628 } while(0)
629# elif defined(QSC_SYSTEM_ARCH_ALPHA)
630# define QSC_SYSTEM_FAST_64X64_MUL(X,Y,Low,High) \
631 do { \
632 asm("umulh %1,%2,%0" : "=r" (*(High)) : "r" (X), "r" (Y)); \
633 *(Low) = (X) * (Y); \
634 } while(0)
635# elif defined(QSC_SYSTEM_ARCH_IA64)
636# define QSC_SYSTEM_FAST_64X64_MUL(X,Y,Low,High) \
637 do { \
638 asm("xmpy.hu %0=%1,%2" : "=f" (*(High)) : "f" (X), "f" (Y)); \
639 *(Low) = (X) * (Y); \
640 } while(0)
641# elif defined(QSC_SYSTEM_ARCH_PPC)
642# define QSC_SYSTEM_FAST_64X64_MUL(X,Y,Low,High) \
643 do { \
644 asm("mulhdu %0,%1,%2" : "=r" (*(High)) : "r" (X), "r" (Y) : "cc"); \
645 *(Low) = (X) * (Y); \
646 } while(0)
647# endif
