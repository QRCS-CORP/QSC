166/*
167 * The falcon_max_bl_small[] and falcon_max_bl_large[] contain the lengths, in 31-bit
168 * words, of intermediate values in the computation:
169 *
170 * falcon_max_bl_small[depth]: length for the input f and g at that depth
171 * falcon_max_bl_large[depth]: length for the unreduced F and G at that depth
172 *
173 * Rules:
174 *
175 * - Within an array, values grow.
176 *
177 * - The 'SMALL' array must have an entry for maximum depth, corresponding
178 * to the size of values used in the binary GCD. There is no such value
179 * for the 'LARGE' array (the binary GCD yields already reduced
180 * coefficients).
181 *
182 * - falcon_max_bl_large[depth] >= falcon_max_bl_small[depth + 1].
183 *
184 * - Values must be large enough to handle the common cases, with some
185 * margins.
186 *
187 * - Values must not be "too large" either because we will convert some
188 * integers into floating-point values by considering the top 10 words,
189 * i.e. 310 bits; hence, for values of length more than 10 words, we
190 * should take care to have the length centered on the expected size.
191 *
192 * The following average lengths, in bits, have been measured on thousands
193 * of random keys (fg = max length of the absolute value of coefficients
194 * of f and g at that depth; FG = idem for the unreduced F and G; for the
195 * maximum depth, F and G are the output of binary GCD, multiplied by q;
196 * for each value, the average and standard deviation are provided).
197 *
198 * Binary case:
199 * depth: 10 fg: 6307.52 (24.48) FG: 6319.66 (24.51)
200 * depth: 9 fg: 3138.35 (12.25) FG: 9403.29 (27.55)
201 * depth: 8 fg: 1576.87 ( 7.49) FG: 4703.30 (14.77)
202 * depth: 7 fg: 794.17 ( 4.98) FG: 2361.84 ( 9.31)
203 * depth: 6 fg: 400.67 ( 3.10) FG: 1188.68 ( 6.04)
204 * depth: 5 fg: 202.22 ( 1.87) FG: 599.81 ( 3.87)
205 * depth: 4 fg: 101.62 ( 1.02) FG: 303.49 ( 2.38)
206 * depth: 3 fg: 50.37 ( 0.53) FG: 153.65 ( 1.39)
207 * depth: 2 fg: 24.07 ( 0.25) FG: 78.20 ( 0.73)
208 * depth: 1 fg: 10.99 ( 0.08) FG: 39.82 ( 0.41)
209 * depth: 0 fg: 4.00 ( 0.00) FG: 19.61 ( 0.49)
210 *
211 * Integers are actually represented either in binary notation over
212 * 31-bit words (signed, using two's complement), or in RNS, modulo
213 * many small primes. These small primes are close to, but slightly
214 * lower than, 2^31. Use of RNS loses less than two bits, even for
215 * the largest values.
216 *
217 * IMPORTANT: if these values are modified, then the temporary buffer
218 * sizes (FALCON_KEYGEN_TEMP_*, in inner.h) must be recomputed
219 * accordingly.
220 */
221extern const size_t falcon_max_bl_small[FALCON_MAX_BL_SMALL_SIZE];

223extern const size_t falcon_max_bl_large[FALCON_MAX_BL_LARGE_SIZE];
